/**
 * The class denotes a layer object that represents one of the several pages in a root document.
 * A 'Pages' object can have a 'Root' object as a parent.
 * A 'Pages' object can have several sections or components as children.
 */
class Page {
    /**
     * The root object of current page
     * @type {Object}
     */
    m_objParent = null;

    /**
     * A flag indicates the animaiton invoked is completed or not
     * @type {boolean}
     */
    m_bAnimationCompleted = true;

    /**
     * A constructor
     * @param {Object} parent
     */
    constructor(parent) {
        this.m_objParent = parent;
    }

    /**
     * A reset function
    */
    reset() {
        this.m_objParent = null;
        this.m_bAnimationCompleted = false;
    }
}
