/**
 * The class denotes a document object of a file that includes several pages.
 * A 'Root' object have no object as a parent.
 * A 'Root' object can have several pages as children.
 */
class Root {
    /**
     * A flag indicates an animaiton is completed or not
     * @type {boolean}
     */
    m_bAnimationCompleted = true;

    /**
     * A constructor
    */
    constructor() {
        this.m_bAnimationCompleted = false;
    }

    /**
     * A reset function
    */
    reset() {
        this.m_bAnimationCompleted = false;
    }
}
